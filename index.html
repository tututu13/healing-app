<script>

  // WABCO 五步法的内容配置

  const steps = [

    {

      letter: "W (Write) - 写下来",

      desc: "把问题写下来，弄清楚你恐惧的具体是什么？把你面对的麻烦和内心的害怕都写在这里。",

      placeholder: "我现在的困境是... 我真正害怕发生的是..."

    },

    {

      letter: "A (Ask) - 寻求",

      desc: "根据你的需要寻求帮助。谁能在这件事上给你情感支持或实用建议？",

      placeholder: "我可以向___寻求帮助，我需要他们帮我..."

    },

    {

      letter: "B (Break) - 分解",

      desc: "将问题分解成几个部分。不要试图一次解决所有麻烦，第一步、第二步分别可以做什么？",

      placeholder: "1. 我今天能做的第一件小事是...\n2. 接下来我可以..."

    },

    {

      letter: "C (Curb) - 抑制",

      desc: "抑制自我批评的欲望。世上没有完美的人，不要在此时评判自己。写下一句安抚“内在小孩”的话。",

      placeholder: "我不会因为感到害怕而羞愧，因为..."

    },

    {

      letter: "O (Optimistically) - 乐观",

      desc: "乐观地想象解决问题后你会有什么感觉。这会让你立刻平静下来。",

      placeholder: "当这一切过去后，我会感到..."

    }

  ];


  let currentStep = 0;

  const userAnswers = [];


  const stepIdEl = document.getElementById('step-id');

  const questionEl = document.getElementById('question-text');

  const descEl = document.getElementById('desc-text');

  const inputEl = document.getElementById('user-input');

  const stepContainer = document.getElementById('step-container');

  const reportContainer = document.getElementById('report-container');

  const reportContent = document.getElementById('report-content');


  function initStep() {

    stepIdEl.innerText = `步骤 ${currentStep + 1} / 5`;

    questionEl.innerText = steps[currentStep].letter;

    descEl.innerText = steps[currentStep].desc;

    inputEl.value = '';

    inputEl.placeholder = steps[currentStep].placeholder;

    if(currentStep === 4) {

      document.getElementById('next-btn').innerText = '生成我的应对方案';

    }

  }


  function nextStep() {

    const answer = inputEl.value.trim();

    if(!answer) {

      alert("请写下一点你的想法，这会帮助你整理思绪哦。");

      return;

    }


    userAnswers.push({

      title: steps[currentStep].letter,

      content: answer

    });


    currentStep++;


    if(currentStep < 5) {

      initStep();

    } else {

      showReport();

    }

  }


  function showReport() {

    stepContainer.style.display = 'none';

    reportContainer.style.display = 'block';


    let html = '';

    userAnswers.forEach(ans => {

      html += `

        <div class="report-item">

          <h3>${ans.title}</h3>

          <p>${ans.content}</p>

        </div>

      `;

    });

    reportContent.innerHTML = html;

  }


  // 初始化第一步

  initStep();

</script>
